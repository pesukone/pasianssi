8   /**
9    * Luokka tarjoaa korttipakan täyttämiseen liittyviä apumetodeja.
10   * @see fi.jussi.pasianssi.kortit.Korttipakka
11   */
12  public class Pakantayttaja {
13      
14  	/**
15  	 * Metodi, joka alustaa tavallisen 52 kortin korttipakan. Parametrina
16  	 * annetun pakan oletetaan olevan tyhjä.
17  	 * @param pakka		alustettava pakka
18  	 */
19  	public static void alustaPeruskorttipakka(Korttipakka pakka) {
20  		luoMaanKortit(pakka, Maa.HERTTA);
21  		luoMaanKortit(pakka, Maa.PATA);
22  		luoMaanKortit(pakka, Maa.RISTI);
23  		luoMaanKortit(pakka, Maa.RUUTU);
24  	}
25      
26  	/**
27  	 * Metodi, joka alustaa 104 kortin pakan, jota Spider-pasianssissa
28  	 * käytetään. Parametrina annetun pakan oletetaan olevan tyhjä.
29  	 * @param pakka		alustettava pakka
30  	 */
31  	public static void alustaTuplakorttipakka(Korttipakka pakka) {
32  		alustaPeruskorttipakka(pakka);
33  		alustaPeruskorttipakka(pakka);
34  	}
35      
36  	private static void luoMaanKortit(Korttipakka pakka, Maa maa) {
37  		for (int i = 1; i <= 13; i++) {
38  			Kortti kortti = new Kortti(maa, i);
39  			pakka.lisaaKortti(kortti);
40  		}
41  	}
42  }
