3   import fi.jussi.pasianssi.kortit.Kortti;
4   
5   /**
6    * Luokka tarjoaa {@link fi.jussi.pasianssi.kortit.Kortti}-olioiden
7    * vertailemisessa käytettäviä apumetodeja.
8    * @see fi.jussi.pasianssi.kortit.Kortti
9    */
10  public class Kortinvertailija {
11  	
12  	/**
13  	 * Metodi kertoo, onko kahden kortin maat samat.
14  	 * @param kortti vertailtava kortti
15  	 * @param toinen toinen vertailtava kortti
16  	 * @return tosi, jos vertailtavien korttien maat ovat samat
17  	 */
18  	public static boolean samaMaa(Kortti kortti, Kortti toinen) {
19  		return kortti.getMaa().equals(toinen.getMaa());
20  	}
21  	
22  	/**
23  	 * Metodi kertoo, onko toinen parametreinä annetuista korteista yhtä
24  	 * pienempi kuin toinen.
25  	 * @param pienempi korteista pienempi
26  	 * @param suurempi korteista suurempi
27  	 * @return tosi, jos ensimmäinen kortti on yhden pienempi kuin toinen
28  	 */
29  	public static boolean yhdenEro(Kortti pienempi, Kortti suurempi) {
30  		return pienempi.getArvo() == suurempi.getArvo() - 1;
31  	}
32  	
33  	/**
34  	 * Metodi kertoo, onko toinen parametreinä annetuista korteista yhtä
35  	 * pienempi kuin toinen, ja ovatko ne samaa maata.
36  	 * @param pienempi korteista pienempi
37  	 * @param suurempi korteista suurempi
38  	 * @return epätosi, jos ensimmäinen kortti ei ole yhtä pienempi kuin toinen,
39  	 * tai kortit ovat eri maata
40  	 */
41  	public static boolean samaaMaataJaYhdenEro(Kortti pienempi, Kortti suurempi) {
42  		return samaMaa(pienempi, suurempi) && yhdenEro(pienempi, suurempi);
43  	}
44  }
