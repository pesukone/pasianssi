68  	private ImageView piirraPakka(Korttipakka pakka) {
69  		ImageView pakkanappi = new ImageView();
70  		pakkanappi.setImage(new Image(getClass().getClassLoader().getResourceAsStream("tausta.jpg")));
71  		pakkanappi.setFitWidth(120);
72  		pakkanappi.setPreserveRatio(true);
73  		BorderPane.setAlignment(pakkanappi, Pos.BOTTOM_RIGHT);
74  		
75  		pakkanappi.setOnMouseClicked(new EventHandler<MouseEvent>() {
76  			@Override
77  			public void handle(MouseEvent event) {
78  				pasianssi.nosta10Korttia();
79  				
80  				for (Korttipino pino : pasianssi.getPinot()) {
81  					ImageView uusi = piirraNakyvaKortti(pino.getNakyvat().hanta());
82  					pinokuvat.get(pino).getChildren().add(uusi);
83  					asetaVali(uusi);
84  					pinokuvat.get(pino).requestLayout();
85  				}
86  				
87  				if (pasianssi.getPakka().tyhja()) {
88  					((BorderPane) pakkanappi.getParent()).getChildren().remove(pakkanappi);
89  				}
90  				event.consume();
91  			}
92  		});
93  		
94  		return pakkanappi;
95  	}

145 		ImageView palautettava = new ImageView();
146 		palautettava.setImage(kortinKuva(kortti.getKortti()));

151 		kuvat.put(kortti, palautettava);
152 		
153 		palautettava.setOnMouseClicked(new EventHandler<MouseEvent>() {
154 			@Override
155 			public void handle(MouseEvent event) {
156 				if (siirrettava == null) {
157 					siirrettava = palautettava;
158 					
159 					for (Korttipino pino : pasianssi.getPinot()) {
160 						if (pino.onPinossa(kortti)) {
161 							lahde = pino;
162 							break;
163 						}
164 					}
165 					
166 					event.consume();
167 				}
168 			}
169 		});
170 		
171 		return palautettava;
172 	}

183 	private AnchorPane piirraKorttipino(Korttipino pino) {
184 		AnchorPane pane = new AnchorPane();
185 		pinot.put(pane, pino);
186 		pinokuvat.put(pino, pane);
187         
188 		Rectangle nakymaton = new Rectangle(120, 120, Color.TRANSPARENT);
189 		pane.getChildren().add(nakymaton);
190 		
191 		for (int i = 0; i < pino.getKaannetyt().size(); i++) {
192 			ImageView kortti = piirraKaannettyKortti();
193 			pane.getChildren().add(kortti);
194 			
195 			asetaVali(kortti);
196 		}
197 
198 		NakyvaKortti nakyva = pino.getNakyvat();
199         
200 		if (nakyva == null) {
201 			return pane;
202 		}
203         
204 		while (nakyva != null) {
205 			ImageView korttikuva = piirraNakyvaKortti(nakyva);
206 			pane.getChildren().add(korttikuva);
207 			asetaVali(korttikuva);
208 			
209 			pane.setOnMouseClicked(new EventHandler<MouseEvent>() {
210 				@Override
211 				public void handle(MouseEvent event) {
212 					if (siirrettava != null) {
213 						if (pino.onPinossa(kortit.get(siirrettava))) {
214 							siirrettava = null;
215 							event.consume();
216 							return;
217 						}
218 						
219 						int kohteenKorttimaara;
220 						if (pino.getNakyvat() == null) {
221 							kohteenKorttimaara = 0;
222 						} else {
223 							kohteenKorttimaara = pino.getNakyvat().korttimaara();
224 						}
225 						
226 						if (lahde.siirraKortti(kortit.get(siirrettava), pino)) {
227 							
228 							while (true) {
229 								((AnchorPane) siirrettava.getParent()).getChildren().remove(siirrettava);
230 								pane.getChildren().add(siirrettava);
231 								asetaVali(siirrettava);
232 								pane.requestLayout();
233 								siirrettava.getParent().requestLayout();
234 								if (kortit.get(siirrettava).getSeuraava() != null) {
235 									siirrettava = kuvat.get(kortit.get(siirrettava).getSeuraava());
236 								} else {
237 									break;
238 								}
239 							}
240 							
241 							if (pino.getNakyvat() == null || kohteenKorttimaara > pino.getNakyvat().korttimaara()) {
242 								for (int i = 1; i <= 13; i++) {
243 									pane.getChildren().remove(pane.getChildren().size() - 1);
244 								}
245 								
246 								if (pino.getNakyvat() != null && pino.getNakyvat().korttimaara() == 1) {
247 									pane.getChildren().remove(pane.getChildren().size() - 1);
248 									ImageView kaannetty = piirraNakyvaKortti(pino.getNakyvat());
249 									pane.getChildren().add(kaannetty);
250 									asetaVali(kaannetty);
251 								}
252 								pane.requestLayout();
253 							}
254 						
255 							if (!lahde.tyhja() && lahde.getNakyvat().korttimaara() == 1) {
256 								ImageView kaannetty = piirraNakyvaKortti(lahde.getNakyvat());
257 								AnchorPane lahdekuva = pinokuvat.get(lahde);
258 							
259 								lahdekuva.getChildren().remove(lahdekuva.getChildren().size() - 1);
260 								lahdekuva.getChildren().add(kaannetty);
261 								asetaVali(kaannetty);
262 							}
263 							
264 							if (pasianssi.voitettu()) {
265 								Dialog dialog = new Dialog();
266 								dialog.setContentText("Voitit pelin!");
267 								
268 								ButtonType uusi = new ButtonType("Uusi peli", ButtonData.OK_DONE);
269 								dialog.getDialogPane().getButtonTypes().add(uusi);
270 								
271 								Optional<ButtonType> result = dialog.showAndWait();
272 								if (result.isPresent() && result.get().getButtonData() == ButtonData.OK_DONE) {
273 									setPasianssi(new Pasianssi(pasianssi.getMaita()));
274 									alustaPoyta();
275 								}
276 							}
277 							
278 							if (pasianssi.havitty()) {
279 								Dialog dialog = new Dialog();
280 								dialog.setContentText("Hävisit pelin");
281 								
282 								ButtonType uusi = new ButtonType("Uusi peli", ButtonData.OK_DONE);
283 								dialog.getDialogPane().getButtonTypes().add(uusi);
284 								
285 								Optional<ButtonType> result = dialog.showAndWait();
286 								if (result.isPresent() && result.get().getButtonData() == ButtonData.OK_DONE) {
287 									setPasianssi(new Pasianssi(pasianssi.getMaita()));
288 									alustaPoyta();
289 								}
290 							}
291 						}
292 						siirrettava = null;
293 						lahde = null;
294 						event.consume();
295 					}
296 				}
297 			});
298 			
299 			nakyva = nakyva.getSeuraava();
300 		}
301         
302 		return pane;
303 	}

305 	private Image kortinKuva(Kortti kortti) {
306 		String maa;
307         
308 		switch (kortti.getMaa()) {
309 			case HERTTA:    maa = "hertta";
310 							break;
311 			case RUUTU:     maa = "ruutu";
312 							break;
313 			case RISTI:     maa = "risti";
314 							break;
315 			case PATA:      maa = "pata";
316 							break;
317 			default:        throw new IllegalArgumentException();
318 		}
319         
320 		String polku = maa + kortti.getArvo() + ".png";
321 		return new Image(getClass().getClassLoader().getResourceAsStream(polku));
322 	}
