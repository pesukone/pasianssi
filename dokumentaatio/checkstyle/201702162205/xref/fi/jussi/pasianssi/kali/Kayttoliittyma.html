41  	private ImageView piirraNakyvaKortti(NakyvaKortti kortti) {
42  		ImageView palautettava = new ImageView();
43  		palautettava.setImage(kortinKuva(kortti.getKortti()));
44  		palautettava.setFitWidth(120);
45  		palautettava.setPreserveRatio(true);
46          
47  		return palautettava;
48  	}

50  	private ImageView piirraKaannettyKortti() {
51  		ImageView palautettava = new ImageView();
52  		palautettava.setImage(new Image(getClass().getClassLoader().getResourceAsStream("black_joker.png")));
53  		palautettava.setFitWidth(120);
54  		palautettava.setPreserveRatio(true);
55  		
56  		return palautettava;
57  	}
58      
59  	private AnchorPane piirraKorttipino(Korttipino pino) {
60  		AnchorPane pane = new AnchorPane();
61          
62  		for (int i = 0; i < pino.getKaannetyt().size(); i++) {
63  			ImageView kortti = piirraKaannettyKortti();
64  			pane.getChildren().add(kortti);
65  			
66  			AnchorPane.setTopAnchor(kortti, (pane.getChildren().size() - 1) * 35.0);
67  		}
68  
69  		NakyvaKortti nakyva = pino.getNakyvat();
70          
71  		if (nakyva == null) {
72  			return pane;
73  		}
74          
75  		while (nakyva != null) {
76  			ImageView korttikuva = piirraNakyvaKortti(nakyva);
77  			pane.getChildren().add(korttikuva);
78  			AnchorPane.setTopAnchor(korttikuva, (pane.getChildren().size() - 1) * 35.0);
79  			nakyva = nakyva.getSeuraava();
80  		}
81          
82  		return pane;
83  	}

85  	private Image kortinKuva(Kortti kortti) {
86  		String maa;
87          
88  		switch (kortti.getMaa()) {
89  			case HERTTA:    maa = "hertta";
90  							break;
91  			case RUUTU:     maa = "ruutu";
92  							break;
93  			case RISTI:     maa = "risti";
94  							break;
95  			case PATA:      maa = "pata";
96  							break;
97  			default:        throw new IllegalArgumentException();
98  		}
99          
100 		String polku = maa + kortti.getArvo() + ".png";
101 		return new Image(getClass().getClassLoader().getResourceAsStream(polku));
102 	}
103     
104 	private HBox piirraKorttipinot() {
105 		HBox laatikko = new HBox(5.0);
106 		laatikko.setAlignment(Pos.CENTER);
107         
108 		List<Korttipino> pinot = pasianssi.getPinot();
109             
110 		for (Korttipino pino : pinot) {
111 			laatikko.getChildren().add(piirraKorttipino(pino));
112 		}
113         
114 		return laatikko;
115 	}
