1   package fi.jussi.pasianssi.kortit;
2   
3   import fi.jussi.pasianssi.apu.Pasianssinalustaja;
4   import java.util.List;
5   import java.util.ArrayList;
6   import java.util.EmptyStackException;
7   
8   /**
9    * Pasianssipeliä kuvaava luokka. Pelissä on {@link fi.jussi.pasianssi.kortit.Korttipakka}
10   * ja kymmenen {@link fi.jussi.pasianssi.kortit.Korttipino}a.
11   * @see fi.jussi.pasianssi.kortit.Korttipakka
12   * @see fi.jussi.pasianssi.kortit.Korttipino
13   */
14  public class Pasianssi {
15  	private Korttipakka pakka;
16  	private List<Korttipino> pinot;
17     
18  	/**
19  	 * Konstruktori konstruktoi.
20  	 */
21  	public Pasianssi() {
22  		this.pakka = new Korttipakka();
23  		this.pinot = new ArrayList();
24  		Pasianssinalustaja.alustaPinot(this.pinot);
25  		Pasianssinalustaja.alustaPakka(this.pakka);
26  		Pasianssinalustaja.taytaPinot(this.pinot, this.pakka);
27  	}
28      
29  	/**
30  	 * Metodi, jolla pasianssin pakasta voi nostaa yhden kortin jokaiseen
31  	 * korttipinoon.
32  	 * @return	nostaminen onnistui
33  	 */
34  	public boolean nosta10Korttia() {
35  		for (int i = 0; i < 10; i++) {
36  			Korttipino pino = this.pinot.get(i);
37  			Kortti nostettava;
38  			
39  			try {
40  				nostettava = this.pakka.nosta();
41  			} catch (EmptyStackException e) {
42  				return false;
43  			}
44  			
45  			pino.lisaaNakyvaKortti(nostettava);
46  		}
47  		
48  		return true;
49  	}
50      
51  	public List<Korttipino> getPinot() {
52  		return this.pinot;
53  	}
54  	
55  	public Korttipakka getPakka() {
56  		return this.pakka;
57  	}
58  }
