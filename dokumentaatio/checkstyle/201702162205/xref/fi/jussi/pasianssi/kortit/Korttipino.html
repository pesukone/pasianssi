1   package fi.jussi.pasianssi.kortit;
2   
3   import java.util.Stack;
4   
5   /**
6    * Näkyvien ja käännettyjen korttien pino pasianssissa. Pohjimmaisten
7    * korttien kuvapuolet voivat olla alaspäin. Kortit, joiden kuvapuolet ovat
8    * ylöspäin, kuvataan linkitettynä listana. Kortit, joiden kuvapuolet ovat
9    * alaspäin, kuvataan pinona.
10   * @see fi.jussi.pasianssi.kortit.NakyvaKortti
11   * @see java.util.Stack
12   */
13  public class Korttipino {
14  	private Stack<Kortti> kaannetytKortit;
15  	private NakyvaKortti nakyvatKortit;
16      
17  	/**
18  	 * Konstruktori konstruktoi näkyvien korttien oletusarvoksi null.
19  	 */
20  	public Korttipino() {
21  		this.kaannetytKortit = new Stack();
22  	}
23      
24  	/**
25  	 * Metodi lisää korttipinoon kortin, jonka kuvapuoli on näkyvissä.
26  	 * @param kortti pinoon lisättävä kortti
27  	 */
28  	public void lisaaNakyvaKortti(Kortti kortti) {
29  		if (this.eiNakyviaKortteja()) {
30  			this.nakyvatKortit = new NakyvaKortti(kortti);
31  		} else {
32  			this.nakyvatKortit.lisaa(kortti);
33  		}
34  	}
35      
36  	/**
37  	 * Metodi lisää korttipinoon kortin, jonka kuvapuoli on käännetty alaspäin.
38  	 * Kutsutaan pasianssipelin alustamisen yhteydessä.
39  	 * @param kortti	pinoon lisättävä kortti
40  	 */
41  	public void lisaaKaannettyKortti(Kortti kortti) {
42  		this.kaannetytKortit.push(kortti);
43  	}
44  	
45  	/**
46  	 * Metodi siirtää pöydällä olevan näkyvän kortin ja sitä seuraavat kortit
47  	 * toiseen korttipinoon.
48  	 * @param kortti siirrettävä kortti
49  	 * @param kohde	korttipino, jonka perään kortti/kortit siirretään
50  	 */
51  	public void siirraKortti(NakyvaKortti kortti, Korttipino kohde) {
52  		if (kohde.eiNakyviaKortteja()) {
53  			kohde.setNakyvat(kortti);
54  			kortti.setEdellinen(null);
55  		} else {
56  			kortti.siirra(kohde.getNakyvat());
57  		}
58          
59  		if (kortti.equals(this.nakyvatKortit)) {
60  			this.nakyvatKortit = null;
61  		}
62  	
63  		if (this.eiNakyviaKortteja()) {
64  			this.kaannaKorttiNakyviin();
65  		}
66  	}
67  	
68  	/**
69  	 * Kertoo, onko korttipino tyhjä.
70  	 * @return korttipino on tyhjä.
71  	 */
72      public boolean tyhja() {
73  		return (this.kaannetytKortit.empty() && this.nakyvatKortit == null);
74      }
75  	
76  	/**
77  	 * Metodi, joka laskee korttipinon korttien yhteismäärän.
78  	 * @return korttien määrä
79  	 */
80  	public int korttimaara() {
81  		return this.kaannetytKortit.size() + this.nakyvatKortit.seuraaviaKortteja();
82  	}
83        	
84  	public Stack<Kortti> getKaannetyt() {
85  		return this.kaannetytKortit;
86  	}
87  	
88  	public NakyvaKortti getNakyvat() {
89  		return this.nakyvatKortit;
90  	}
91  	
92  	public void setNakyvat(NakyvaKortti kortti) {
93  		this.nakyvatKortit = kortti;
94  	}
95  
96  	private void kaannaKorttiNakyviin() {
97  		if (!this.kaannetytKortit.isEmpty()) {
98  			this.lisaaNakyvaKortti(this.kaannetytKortit.pop());
99  		}
100 	}
101 	
102 	private boolean eiNakyviaKortteja() {
103 		return this.nakyvatKortit == null;
104 	}
105 }
